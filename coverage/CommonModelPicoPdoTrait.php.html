35     * Combines `prepare` & `execute` in a single function and returns the PDO statement.
36     *
37     *  - When a parameter contains an **array**, it is expanded into multiple placeholders (for `WHERE IN`).
38     *
39     * **Examples:**
40     * ```

44     * // Named placeholders

46     *
47     * // Multiple values for WHERE IN (named placeholders only)
48     * $ids = [1, 2, 3];
49     * $stmt = $db->prepExec('SELECT * FROM users WHERE id IN (:ids)', [':ids' => $ids]);
50     * // SQL: SELECT * FROM users WHERE id IN (:ids0, :ids1, :ids2)
51     * // Params: ["ids0" => 1, "ids1" => 2, "ids2" => 3]

57     * @throws PDOException If the query fails to execute.

60    {
61        if (array_filter($params, '\is_array')) {

263    /**
264     * Expands named placeholders in SQL queries for `IN` clauses.
265     *
266     * - Replaces named placeholders (`:key`) with multiple placeholders (`:key0, :key1, :key2`) for `WHERE IN` queries.
267     * - Updates the bind parameters to match the expanded placeholders.
268     * - Supports **only named placeholders** for arrays (positional `?` placeholders are not expanded).
269     * **Examples:**
270     * ```
271     * $sql = 'SELECT * FROM users WHERE id IN (:ids)';
272     * $params = ['ids' => [1, 2, 3]];
273     *
274     * [$newSql, $newParams] = $this->buildInQuery($sql, $params);
275     *
276     * // Resulting SQL: "SELECT * FROM users WHERE id IN (:ids0, :ids1, :ids2)"
277     * // Resulting Params: [":ids0" => 1, ":ids1" => 2, ":ids2" => 3]
278     * ```
279     *
280     * @param string $sql The SQL query containing named placeholders.
281     * @param array<string, mixed> $params The parameters to bind, where array values are expanded.
282     * @return array{string, array<string, mixed>} The modified SQL query and the updated bind parameters.
283     */
284    protected function buildInQuery(string $sql, array $params): array
285    {
286        if (empty($sql) || empty($params)) {
287            return [$sql, $params];
288        }
289
290        $expandedParams = [];
291
292        foreach ($params as $key => $value) {
293            if (is_array($value) && !is_numeric($key) && !empty($value)) {
294
295                // Only support named parameters for array values
296                $searchKey = str_starts_with($key, ':') ? $key : ":{$key}";
297
298                $valuesIn = array_values($value); // Ensure values are indexed
299
300                $placeholders = array_map(static fn($index) => "{$searchKey}{$index}", array_keys($valuesIn));
301
302                // Match :key followed by a word boundary to prevent partial matches
303                // This ensures :ids doesn't match :ids_extra or :ids2
304                $pattern = '/(?<!:)' . preg_quote($searchKey, '/') . '\b/';
305
306                $sql = preg_replace($pattern, implode(',', $placeholders), $sql);
307
308                $expandedParams = [...$expandedParams, ...array_combine($placeholders, $valuesIn)];
309
310                // Remove original array key
311                unset($params[$key]);
312            } else {
313                $expandedParams[(string)$key] = $value;
314            }
315        }
316
317        return [$sql, $expandedParams];
318    }

333     *
334     * For UPDATE/DELETE operations, if no WHERE clause is provided,
335     * an invalid `WHERE` clause (`WHERE `) will be returned, triggering a PDO exception

337     *

355     * ```

357     * // WHERE email = :where_0 AND status != :where_1
358     * // [':where_0' => 'john@example.com', ':where_1' => 'inactive']
359     * ```
360     *
361     * 4. Raw SQL with named placeholders
362     * ```
363     * buildWhereQuery('role = :role AND status = :status', [':role' => 'admin', ':status' => 'active']);
364     * // WHERE role = :role AND status = :status
365     * // [':role' => 'admin', ':status' => 'active']
366     * ```
367     *
368     * 5. Raw SQL with `IN (:placeholder)` and array binding
369     * ```
370     * buildWhereQuery('id IN (:ids)', [':ids' => [1, 2, 3]]);
371     * // WHERE id IN (:ids0, :ids1, :ids2)
372     * // [':ids0' => 1, ':ids1' => 2, ':ids2' => 3]
373     * ```
374     *
375     * 6. Raw SQL with no bindings
376     * ```
377     * buildWhereQuery('created_at IS NULL');
378     * // WHERE created_at IS NULL
379     * // []
380     * @param string|array<string, mixed> $where Column name, condition string, or associative array
381     * @param int|string|array<string|int>|null $bindings Value for single column or array of bound values for custom condition
